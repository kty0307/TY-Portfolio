5.1 함수 구현 코드¶
In [ ]:
def getListOfPrices(price:int, returns:list)->list:
    price_list = []
    price_virtual = price
    for r in returns:
        #  NaN   (r=0,  )  
        if np.isnan(r):
            r = 0
        #    100   floor   
        price_virtual = math.floor((price_virtual * math.exp(r))/100)*100 
        price_list.append(price_virtual)
    return price_list
